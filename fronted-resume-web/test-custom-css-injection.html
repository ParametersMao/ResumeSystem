🔍 测试自定义CSS注入功能

问题分析

调试信息显示 customCss长度: 0，说明模板数据中没有自定义CSS。

我们需要测试两种情况：

  1. 在模板数据中添加 customCss
  2. 在简历数据中添加 customCss
测试方法1：在简历数据中添加customCss

在浏览器控制台中运行以下代码：

// 获取Vue应用实例 const app = document.querySelector('#app'); if (app && app.__vue__) { const vueApp = app.__vue__; // 查找ResumeEditorNew组件 let resumeEditor = null; function findResumeEditor(component) { if (component.type && component.type.name === 'ResumeEditorNew') { resumeEditor = component; return; } if (component.subTree && component.subTree.component) { findResumeEditor(component.subTree.component); } if (component.children) { component.children.forEach(child => { if (child.component) { findResumeEditor(child.component); } }); } } findResumeEditor(vueApp); if (resumeEditor) { console.log('找到ResumeEditorNew组件'); // 添加自定义CSS到简历数据 if (resumeEditor.setupState && resumeEditor.setupState.resumeData) { resumeEditor.setupState.resumeData.value.customCss = ` .dynamic-resume-preview { border: 5px solid red !important; background: linear-gradient(45deg, #ff6b6b, #4ecdc4) !important; } .resume-content { background: linear-gradient(135deg, #f5f7fa 0%, #c3cfe2 100%) !important; border: 3px solid blue !important; border-radius: 20px !important; } `; console.log('✅ 已添加自定义CSS到简历数据'); } else { console.log('❌ 无法访问简历数据'); } } else { console.log('❌ 未找到ResumeEditorNew组件'); } } else { console.log('❌ 未找到Vue应用'); }
测试方法2：检查模板数据结构

在浏览器控制台中运行以下代码：

// 检查模板数据结构 const app = document.querySelector('#app'); if (app && app.__vue__) { const vueApp = app.__vue__; // 查找ResumeEditorNew组件 let resumeEditor = null; function findResumeEditor(component) { if (component.type && component.type.name === 'ResumeEditorNew') { resumeEditor = component; return; } if (component.subTree && component.subTree.component) { findResumeEditor(component.subTree.component); } if (component.children) { component.children.forEach(child => { if (child.component) { findResumeEditor(child.component); } }); } } findResumeEditor(vueApp); if (resumeEditor && resumeEditor.setupState) { console.log('模板数据结构:', resumeEditor.setupState.templateData); console.log('简历数据结构:', resumeEditor.setupState.resumeData); } }
